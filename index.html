In the mathematical field of graph theory, a bipartite graph (or bigraph) is a graph whose vertices can be divided into two disjoint and independent sets {\displaystyle U}U and {\displaystyle V}V such that every edge connects a vertex in {\displaystyle U}U to one in {\displaystyle V}V.                 for (int st = 0; st < n; ++st) {
                    if (side[st] == -1) {
                        q.push(st);
                        side[st] = 0;
                        while (!q.empty()) {
                            int v = q.front();
                            q.pop();
                            for (int u : adj[v]) {
                                if (side[u] == -1) {
                                    side[u] = side[v] ^ 1;
                                    q.push(u);
                                } else {
                                    is_bipartite &= side[u] != side[v];
                                }
                            }
                        }
                    }
                }
